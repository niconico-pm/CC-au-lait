Create Database BST_ScoreTool;
Use BST_ScoreTool;
Create Table
user(
	UID		serial,
	UserName	char(32) Unique NOT NULL,
	PassHash	char(128) NOT NULL,
	PassSalt	char(32) NOT NULL,

	Primary Key(UID)
);

Create Table
music(
	MusicID		int unsigned,
	Name		varchar(64) NOT NULL,

	Primary Key(MusicID)
);

Create Table
score(
	UID		bigint unsigned,
	MusicID		int unsigned,
	Difficulty	int NOT NULL,
	Score		int unsigned,
	Medal		int,

	Primary Key(UID, MusicID, Difficulty),
	Foreign Key(UID) References user(UID),
	Foreign Key(MusicID) References music(MusicID)
);
